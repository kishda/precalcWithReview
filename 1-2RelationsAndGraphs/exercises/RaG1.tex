\begin{exercise}

Give the Cartesian coordinates for each point on the graph:

\begin{image}
\begin{tikzpicture}
    \begin{axis}[xlabel={},ylabel={},
                xmin=-10,xmax=10,
                ymin=-10,ymax=10,
                xtick={-8,-6,...,8},
                ytick={-8,-6,...,8},
                clip=false]
        \addplot[soliddot] coordinates {(6,6)} node[below]{\large A};
        \addplot[soliddot] coordinates {(-7, 0)} node[above]{\large B};
        \addplot[soliddot] coordinates {(0, -4)} node[right]{\large C};
        \addplot[soliddot] coordinates {(7, 0)} node[above]{\large D};
        \addplot[soliddot] coordinates {(-6, -6)} node[below]{\large E};
        \addplot[soliddot] coordinates {(8, -5)} node[below]{\large F};
	\addplot[soliddot] coordinates {(-2, 8)} node[below]{\large G};	
    \end{axis}
\end{tikzpicture}
\end{image}

\begin{prompt}
\begin{align*}
A&=\left(\answer{6},\answer{6}\right)\\
B&=\left(\answer{-7},\answer{0}\right)\\
C&=\left(\answer{0},\answer{-4}\right)\\
D&=\left(\answer{7},\answer{0}\right)\\
E&=\left(\answer{-6},\answer{-6}\right)\\
F&=\left(\answer{8},\answer{-5}\right)\\
G&=\left(\answer{-2},\answer{8}\right)\\
\end{align*}
\end{prompt}

\end{exercise}
