\begin{exercise}
For each given point, say whether it is a member of the relation given by $x^2 - y^2 = 1$. 

\begin{enumerate}
\item Is $(1, -2)$  in the relation? 
\begin{multipleChoice}
\choice{\text{Yes}}
\choice[correct]{\text{No}}
\end{multipleChoice}
\item Is $(1, 0)$  in the relation? 
\begin{multipleChoice}
\choice[correct]{\text{Yes}}
\choice{\text{No}}
\end{multipleChoice}
\item Is $(0, -1)$  in the relation? 
\begin{multipleChoice}
\choice{\text{Yes}}
\choice[correct]{\text{No}}
\end{multipleChoice}

\end{enumerate}



\end{exercise}
