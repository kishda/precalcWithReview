\begin{exercise}
In each part, an invertible function $f$ will be defined. For each function, find a formula for $f^{-1}$ where $f$ is defined.

\begin{enumerate}

\item $f(x) = 2x - 6$

$f^{-1}(x) = \answer{\frac{6 + x}{2}}$

\item 
$f(x) = 29 - x$

$f^{-1}(x) = \answer{29 - x}$

\item 
$f(x) = \frac{x - 3}{2} + 3$

$f^{-1}(x) = \answer{2x - 3}$

\item $f(x) = \sqrt{5x - 1} + 3$

$f^{-1}(x) = \answer{\frac{(x - 3)^2 + 1}{5}}$
\end{enumerate}

\end{exercise}
