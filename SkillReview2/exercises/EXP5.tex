\begin{exercise}
Which of the following are equivalent to $\frac{y^{2}z^{-3}}{x^{-4}}$?
\begin{selectAll}
\choice {$\frac{y^{-2}z^3}{x^4}$}
\choice[correct] {$\frac{x^4}{z^{3}y^{-2}}$}
\choice[correct]  {$\frac{y^{3}z^{-1}x^{6}}{x^{2}yz^{2}}$}
\choice  {$\frac{y^{-1x^{2}}z^{-2}}{y^{1}x^{-2}z^{-1}}$}
\choice[correct]  {${y^{2}z^{-3}}{x^{4}}$}
\end{selectAll}
\end{exercise}
