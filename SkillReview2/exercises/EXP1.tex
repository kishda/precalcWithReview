\begin{exercise}
Which of the following expressions are equivalent to $n^5$?
  \begin{selectAll}
    \choice{$n^3+n^2$}
    \choice[correct]{$\frac{1}{n^{-5}}$}
    \choice[correct]{$\frac{n^8}{n^3}$}
\choice{$n^5\cdot n^1$}
    \choice[correct]{$\frac{n^3}{n^{-2}}$}
\choice{$n^7-n^2$}
  \end{selectAll}
\end{exercise}
