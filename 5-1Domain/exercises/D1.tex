\begin{exercise}
	The following is the graph of which exponential function?
	\begin{center}
		\begin{tikzpicture}
			\begin{axis}[
				xmin=-4.5, xmax=4.5, ymin=-4.5,ymax=2.1,    
				axis lines =middle, 
				every axis y label/.style={at=(current axis.above origin),anchor=south},
				every axis x label/.style={at=(current axis.right of origin),anchor=west},
				xtick={-5,...,5}, ytick={-4,...,2},
				grid=major, width=4in,
				]
				\addplot[color=red, very thick, smooth, domain=-5:5]{-2^x+1};
			\end{axis}
		\end{tikzpicture}
	\end{center}
	
	\begin{multipleChoice}
		\choice[correct]{$-2^x+1$}
		\choice{$\left(\frac{1}{2}\right)^x - 1$}
		\choice{$2^{-x}+1$}
		\choice{$\left(\frac{1}{2}\right)^{-x}-1$}
	\end{multipleChoice}
\end{exercise}
