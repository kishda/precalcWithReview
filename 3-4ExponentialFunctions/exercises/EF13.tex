\begin{exercise}

  If $y = 4e^{7x}-8$, solve for $x$ in terms of $y$.

  Answer: $x = \frac{\answer{1}}{\answer{7}} \ln \left(\frac{\answer{y+8}}{\answer{4}}\right)$.

\end{exercise}
