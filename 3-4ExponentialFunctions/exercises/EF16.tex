\begin{exercise}

  If $y = -6e^{5x}+3$, solve for $x$ in terms of $y$.

  Answer: $x = \frac{\answer{1}}{\answer{5}} \ln \left(\frac{\answer{3-y}}{\answer{6}}\right)$.

\end{exercise}
