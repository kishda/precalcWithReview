\begin{exercise}

  If $y = 10e^{3x}+5$, solve for $x$ in terms of $y$.

  Answer: $x = \frac{\answer{1}}{\answer{3}} \ln \left(\frac{\answer{y-5}}{\answer{10}}\right)$.

\end{exercise}
