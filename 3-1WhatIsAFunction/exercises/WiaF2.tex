\begin{exercise}
Select all of the relations which represent  $y$ as a function of $x$.
\begin{selectAll}
\choice[correct]{\{$(-3, 9)$, $\;(-2, 4)$, $\;(-1, 1)$, $\;(0, 0)$, $\;(1, 1)$, $\;(2, 4)$, $\;(3, 9)\}$ }
\choice{$\left\{ (-3,0), (1,6), (2, -3), (4,2), (-5,6), (4, -9), (6,2) \right\}$}
\choice[correct]{$\left\{ (-3,0), (-7,6), (5,5), (6,4), (4,9), (3,0) \right\}$}
\choice[correct]{$\left\{ (1,2), (4,4), (9,6), (16,8), (25,10), (36, 12), \ldots \right\}$}
\choice{\{($x, y) \, | \, x$ is an odd integer, and $y$ is an even integer\}}
\choice[correct]{\{$(x, 1) \, | \, x$ is an irrational number\}}
\choice[correct]{\{$(1, 0)$, $\;(2, 1)$, $\;(4, 2)$, $\;(8, 3)$, $\;(16, 4)$, $\;(32, 5), \;$ \ldots\}}
\choice[correct]{\{$\ldots, \; (-3, 9)$, $\;(-2, 4)$, $\;(-1, 1)$, $\;(0, 0)$, $\;(1, 1)$, $\;(2, 4)$, $\;(3, 9), \;$ \ldots\}}
\choice{$\{ (-2, y) \, | \, -3 < y < 4\}$}
\choice[correct]{$\{ (x,3) \, | \,  -2 \leq x < 4\}$}
\choice[correct]{$\{ \left(x,x^2\right) \, | \, \text{$x$ is a real number} \}$}
\choice{$\{ \left(x^2,x\right) \, | \, \text{$x$ is a real number} \}$}
\end{selectAll}

\end{exercise}
