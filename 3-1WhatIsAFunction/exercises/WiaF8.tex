Let $f(t)$ denote the number of people eating in a restaurant $t$ minutes after 5 PM. Answer the following questions:

\begin{exercise}
Which of the following statements best describes the significance of the expression $f(4)=21$?

\begin{multipleChoice}
\choice{There are 4 people eating at 5:21 PM}
\choice[correct]{There are 21 people eating at 5:04 PM}
\choice{There are 21 people eating at 9:00 PM}
\choice{Every 4 minutes, 21 more people are eating}
\choice{None of the above}
\end{multipleChoice}
\end{exercise}

\begin{exercise}
Which of the following statements best describes the significance of the expression $f(a)=20$?

\begin{multipleChoice}
\choice[correct]{$a$ minutes after 5 PM there are 20 people eating}
\choice{Every 20 minutes, the number of people eating has increased by $a$ people}
\choice{At 5:20 PM there are $a$ people eating}
\choice{$a$ hours after 5 PM there are 20 people eating}
\choice{None of the above}
\end{multipleChoice}
\end{exercise}

\begin{exercise}
Which of the following statements best describes the significance of the expression $f(20)=b$?

\begin{multipleChoice}
\choice{Every 20 minutes, the number of people eating has increased by $b$ people}
\choice{$b$ minutes after 5 PM there are 20 people eating}
\choice[correct]{At 5:20 PM there are $b$ people eating}
\choice{$b$ hours after 5 PM there are 20 people eating}
\choice{None of the above}
\end{multipleChoice}
\end{exercise}

\begin{exercise}
Which of the following statements best describes the significance of the expression $n=f(t)$?

\begin{multipleChoice}
\choice{Every $t$ minutes, $n$ more people have begun eating}
\choice{$n$ hours after 5 PM there are $t$ people eating}
\choice{$n$ minutes after 5 PM there are $t$ people eating}
\choice{$t$ hours after 5 PM there are $n$ people eating}
\choice[correct]{None of the above}
\end{multipleChoice}
\end{exercise}
